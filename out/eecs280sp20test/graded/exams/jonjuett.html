// You may assume type T supports ==, !=, <, <=, >, and >=
template <typename T>
class MinQueue { 
public:
  MinQueue() : size(0) { } // default ctor

  // REQUIRES: the MinQueue is not empty
  // EFFECTS: Returns the minimum item, but
  // does not remove it from the MinQueue
  const T & min() const;

  // REQUIRES: the MinQueue is not empty
  // EFFECTS: Removes the minimum item from the MinQueue
  void popMin();

  // REQUIRES: the MinQueue is not full
  // EFFECTS: Adds 'item' to the MinQueue
  void push(const T &item);

private:
  // For simplicity, we assume a fixed capacity
  // and that the user of a MinQueue never exceeds
  // this. This isn't a dynamic memory question :).
  int data[10];

  // number of valid items in the data array
  int size;
};

Here's an example of using a MinQueue:

int main() {
  MinQueue<int> q;
  q.push(6);
  q.push(2);
  q.push(5);

  cout << q.min(); // prints 2

  q.popMin(); // removes 2
  cout << q.min(); // prints 5
}

The operator should be implemented as a non-member function and should remove the minimum element (you may call popMin() in your implementation). The operator should evaluate back into the MinQueue object, so that multiple -- operations may be chained.

For example:

int main() {
  MinQueue<int> q;
  q.push(6);
  q.push(2);
  q.push(5);
  q.push(10);
  cout << q.min(); // prints 2

  --q; // removes 2
  cout << q.min(); // prints 5

class Primes {
public:
  // Constructor
  Primes(int num)
   : num(num), data(new int[num]) {
    // fills data array with the
    // first 'num' prime numbers
  }

  // Copy Constructor
  Primes(const Primes &other)
   : num(other.num), data(new int[other.num]) {
    for(int i = 0; i < num; ++i) {
      data[i] = other.data[i];
    }
  }

  // Destructor
  ~Primes() {
    delete[] data;
  }

  // Assignment operator, to be implemented...
  Primes & operator=(const Primes &rhs);

private:
  int num;
  int *data;

};
